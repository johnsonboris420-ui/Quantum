# =============================================
# 1. QUANTUM TESTING FRAMEWORK (from original)
# =============================================
class QuantumTestingFramework:
    def __init__(self):
        self.cycle = 0

    def run_adaptive_test_cycle_once(self):
        self.cycle += 1
        entanglement = 0.92 + 0.08 * (self.cycle % 6)
        error = max(0.0, 0.018 - 0.0005 * self.cycle)
        return f"Cycle #{self.cycle} | Entanglement: {entanglement:.3f} | Error: {error:.4f}"

    def calculate_quantum_cooldown(self):
        return round(max(4.5 - 0.15 * (self.cycle // 3), 1.2), 1)


# =============================================
# 2. PURE-PYTHON 4D MANDELBULB SCALAR
# =============================================
def mandelbulb_4d_power(p, n=8, max_iter=50, bailout=4.0):
    """Exact copy of your latest scalar function – perfect for distance estimation or testing."""
    cx, cy, cz, cw = p
    x = y = z = w = 0.0
    for i in range(max_iter):
        r_xy = math.hypot(x, y)
        r_xyz = math.hypot(r_xy, z)
        r = math.hypot(r_xyz, w)
        if r > bailout:
            if n == 1:
                return i + 1.0
            return i + 1.0 - math.log(math.log(r) / math.log(bailout)) / math.log(n)
        if r == 0.0:
            phi = theta = psi = 0.0
        else:
            phi = math.atan2(y, x)
            theta = math.atan2(z, r_xy)
            psi = math.atan2(w, r_xyz)
        r_new = r ** n
        phi_new = n * phi
        theta_new = n * theta
        psi_new = n * psi
        cos_psi = math.cos(psi_new)
        sin_psi = math.sin(psi_new)
        cos_theta = math.cos(theta_new)
        sin_theta = math.sin(theta_new)
        cos_phi = math.cos(phi_new)
        sin_phi = math.sin(phi_new)
        x = r_new * cos_psi * cos_theta * cos_phi + cx
        y = r_new * cos_psi * cos_theta * sin_phi + cy
        z = r_new * cos_psi * sin_theta + cz
        w = r_new * sin_psi + cw
    return float(max_iter)


# =============================================
# 3. NUMPY VECTORIZED RENDERER (fast grid mode)
# =============================================
def render_4d_slice(power=8, max_iter=30, bailout=4.0, res=400,
                    w_slice=0.0, bound=1.5, mode="2d"):
    """Returns either 2D escape-time array or 3D surface points."""
    xs = np.linspace(-bound, bound, res)
    ys = np.linspace(-bound, bound, res)
    if mode == "2d":
        cx, cy = np.meshgrid(xs, ys, indexing='ij')
        c_flat = np.stack([cx.ravel(), cy.ravel(),
                           np.zeros(res*res), np.full(res*res, w_slice)], axis=-1)
        # Use scalar in loop for simplicity (fast enough at 400²)
        escape = np.zeros(res*res)
        for i in range(res*res):
            escape[i] = mandelbulb_4d_power(c_flat[i], power, max_iter, bailout)
        return escape.reshape(res, res).T
    else:
        # 3D surface mode (w=0 slice)
        zs = np.linspace(-bound, bound, res//2)
        cx, cy, cz = np.meshgrid(xs, ys, zs, indexing='ij')
        c_flat = np.stack([cx.ravel(), cy.ravel(), cz.ravel(),
                           np.full(cx.size, w_slice)], axis=-1)
        inside = np.zeros(cx.size, dtype=bool)
        for i in range(cx.size):
            if mandelbulb_4d_power(c_flat[i], power, max_iter//2, bailout) == max_iter//2:
                inside[i] = True
        inside = inside.reshape(res, res, res//2)
        # Simple surface extraction
        surface = np.where(inside & ~np.roll(inside, 1, axis=0) |
                           inside & ~np.roll(inside, 1, axis=1) |
                           inside & ~np.roll(inside, 1, axis=2))
        return (xs[surface[0]], ys[surface[1]], zs[surface[2]])


# =============================================
# 4. TIMELAPSE & FRAME SAVER
# =============================================
def save_frame(img_array, cycle):
    os.makedirs("mandelbulb_frames", exist_ok=True)
    filename = f"mandelbulb_frames/mandelbulb_cycle_{cycle:04d}.png"
    Image.fromarray(img_array.astype(np.uint8)).save(filename)
    return filename


def create_timelapse_video(framerate=15):
    try:
        cmd = [
            "ffmpeg", "-y", "-framerate", str(framerate),
            "-i", "mandelbulb_frames/mandelbulb_cycle_%04d.png",
            "-c:v", "libx264", "-pix_fmt", "yuv420p",
            "-preset", "fast", "-crf", "18",
            "mandelbulb_4d_timelapse.mp4"
        ]
        subprocess.run(cmd, stdout=subprocess.DEVNULL, stderr=subprocess.DEVNULL, check=True)
        return "✅ Timelapse saved: mandelbulb_4d_timelapse.mp4"
    except Exception as e:
        return f"⚠️ FFmpeg not found: {e} (video skipped)"


# =============================================
# 5. MAIN TKINTER APP
# =============================================
class Mandelbulb4DApp:
    def __init__(self):
        self.root = tk.Tk()
        self.root.title("4D Mandelbulb Explorer – Quantum Edition")
        self.root.geometry("1280x900")
        self.root.configure(bg="#0a0a0a")

        self.qtf = QuantumTestingFramework()
        self.cycle = 0
        self.running = False

        # === LEFT PANEL: CONTROLS ===
        ctrl = tk.Frame(self.root, bg="#0a0a0a")
        ctrl.pack(side="left", fill="y", padx=15, pady=15)

        tk.Label(ctrl, text="4D MANDELBULB APP", fg="#00ffcc", bg="#0a0a0a",
                 font=("Helvetica", 18, "bold")).pack(pady=10)

        # Sliders
        self.power_var = tk.DoubleVar(value=8.0)
        tk.Label(ctrl, text="Power (n)", fg="white", bg="#0a0a0a").pack(anchor="w")
        ttk.Scale(ctrl, from_=2, to=20, variable=self.power_var, orient="horizontal").pack(fill="x")

        self.iter_var = tk.IntVar(value=35)
        tk.Label(ctrl, text="Max Iterations", fg="white", bg="#0a0a0a").pack(anchor="w")
        ttk.Scale(ctrl, from_=10, to=100, variable=self.iter_var, orient="horizontal").pack(fill="x")

        self.res_var = tk.IntVar(value=350)
        tk.Label(ctrl, text="Resolution", fg="white", bg="#0a0a0a").pack(anchor="w")
        ttk.Scale(ctrl, from_=100, to=800, variable=self.res_var, orient="horizontal").pack(fill="x")

        self.w_var = tk.DoubleVar(value=0.0)
        tk.Label(ctrl, text="W-Slice (4th dimension)", fg="white", bg="#0a0a0a").pack(anchor="w")
        ttk.Scale(ctrl, from_=-1.5, to=1.5, variable=self.w_var, orient="horizontal").pack(fill="x")

        self.bound_var = tk.DoubleVar(value=1.5)
        tk.Label(ctrl, text="Bounds", fg="white", bg="#0a0a0a").pack(anchor="w")
        ttk.Scale(ctrl, from_=0.5, to=3.0, variable=self.bound_var, orient="horizontal").pack(fill="x")

        # Buttons
        btn_frame = tk.Frame(ctrl, bg="#0a0a0a")
        btn_frame.pack(pady=20)

        ttk.Button(btn_frame, text="Render 2D Slice", command=self.render_2d).grid(row=0, column=0, padx=5, pady=5)
        ttk.Button(btn_frame, text="Render 3D Surface", command=self.render_3d).grid(row=0, column=1, padx=5, pady=5)
        ttk.Button(btn_frame, text="Start Quantum Timelapse", command=self.start_timelapse).grid(row=1, column=0, columnspan=2, pady=5)
        ttk.Button(btn_frame, text="Stop", command=self.stop).grid(row=1, column=2, padx=5)

        self.status = tk.Label(ctrl, text="Ready – Press Render", fg="#00ffcc", bg="#0a0a0a", font=("Helvetica", 10))
        self.status.pack(pady=10)

        # === RIGHT PANEL: DISPLAY ===
        self.fig, self.ax = plt.subplots(figsize=(8, 8), facecolor="#0a0a0a")
        self.ax.set_facecolor("#0a0a0a")
        self.canvas = FigureCanvasTkAgg(self.fig, master=self.root)
        self.canvas.get_tk_widget().pack(side="right", fill="both", expand=True)

        self.root.protocol("WM_DELETE_WINDOW", self.on_close)

    def render_2d(self):
        self.status.config(text="Rendering 2D slice...")
        self.root.update()
        power = self.power_var.get()
        iters = self.iter_var.get()
        res = self.res_var.get()
        w = self.w_var.get()
        bound = self.bound_var.get()

        esc = render_4d_slice(power, iters, 4.0, res, w, bound, mode="2d")
        self.ax.clear()
        self.ax.imshow(esc, cmap="inferno", extent=[-bound, bound, -bound, bound], origin="lower")
        self.ax.set_title(f"4D Mandelbulb (w={w:.2f}) – Power {power} | Iter {iters}")
        self.ax.set_xlabel("X"); self.ax.set_ylabel("Y")
        self.canvas.draw()

        # Also save frame
        img = (esc / esc.max() * 255).astype(np.uint8)
        colored = plt.cm.inferno(img)[:, :, :3] * 255
        save_frame(colored.astype(np.uint8), self.cycle)
        self.status.config(text=f"2D rendered & saved (cycle {self.cycle})")

    def render_3d(self):
        self.status.config(text="Rendering 3D surface (may take a few seconds)...")
        self.root.update()
        power = self.power_var.get()
        iters = self.iter_var.get()
        res = self.res_var.get() // 2
        w = self.w_var.get()
        bound = self.bound_var.get()

        try:
            x, y, z = render_4d_slice(power, iters, 4.0, res, w, bound, mode="3d")
            self.ax.clear()
            self.ax = self.fig.add_subplot(111, projection='3d')
            self.ax.scatter(x, y, z, c=z, cmap="inferno", s=0.8)
            self.ax.set_title(f"3D Slice of 4D Mandelbulb (w={w:.2f})")
            self.canvas.draw()
            self.status.config(text="3D surface rendered")
        except Exception as e:
            self.status.config(text=f"3D error: {str(e)[:60]}")

    def start_timelapse(self):
        if self.running:
            return
        self.running = True
        threading.Thread(target=self._timelapse_thread, daemon=True).start()

    def _timelapse_thread(self):
        self.cycle = 0
        while self.running and self.cycle < 120:  # safety limit
            self.qtf.run_adaptive_test_cycle_once()
            self.status.config(text=self.qtf.run_adaptive_test_cycle_once())
            self.root.update()

            # Render current w slice (w slowly changes)
            w = -1.2 + (self.cycle / 30) * 2.4
            self.w_var.set(w)
            esc = render_4d_slice(self.power_var.get(),
                                  self.iter_var.get(),
                                  4.0,
                                  self.res_var.get(),
                                  w,
                                  self.bound_var.get(),
                                  mode="2d")
            img = (esc / esc.max() * 255).astype(np.uint8)
            colored = (plt.cm.inferno(img)[:, :, :3] * 255).astype(np.uint8)
            save_frame(colored, self.cycle)

            self.ax.clear()
            self.ax.imshow(esc, cmap="inferno", extent=[-1.5, 1.5, -1.5, 1.5], origin="lower")
            self.ax.set_title(f"Quantum Cycle {self.cycle} | w = {w:.3f}")
            self.canvas.draw()

            if self.cycle % 10 == 0:
                msg = create_timelapse_video()
                self.status.config(text=msg)

            self.cycle += 1
            time.sleep(self.qtf.calculate_quantum_cooldown() * 0.3)  # faster for demo

        self.running = False
        self.status.config(text="Timelapse finished!")

    def stop(self):
        self.running = False

    def on_close(self):
        self.running = False
        self.root.destroy()


if __name__ == "__main__":
    print("🌌 Launching 4D Mandelbulb Explorer App...")
    app = Mandelbulb4DApp()
    app.root.mainloop()
